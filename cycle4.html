5.Write a C program to calculate the sum of 1! + 2! + 3! + ..... + n!.

COODE: 
                #include 

                    int main(){
                        int n,product=1,sum=0,i;
                        printf("Enter n value : ");
                        scanf("%d",&n);
                        i = 1;
                        while(i<=n){
                            product *=i;
                            sum = sum + product;
                            i++;
                        }
                        printf("Sum of series : %d\n",sum);
                    }
