3.Write a program to find the sum of individual digits of a given number.

COODE: 
            #include 

                int main(){
                    int sum=0,n,lastdigit;
                    printf("Enter an integer : ");
                    scanf("%d",&n);
                    printf("The sum of digits of the given number %d ",n);
                    while(n!=0){
                        lastdigit = n % 10;
                        sum += lastdigit;
                        n/=10;
                    }
                    printf("= %d\n",sum);
                    
                    
                }
